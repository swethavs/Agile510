001package edu.pdx.cs410J.pmudgal;

003import edu.pdx.cs410J.AbstractAppointmentBook;
004
005import java.util.ArrayList;
006import java.util.Collection;
007
008/**
009 * Created by Vijay on 6/27/2016.
010 */
011public class AppointmentBook extends AbstractAppointmentBook<Appointment> {
012
013    private ArrayList<Appointment> appointments = new ArrayList<Appointment>();
014
015    private String ownerName;
016
017    @Override
018    public String getOwnerName() {
019        return ownerName;
020    }
021
022    @Override
023    public Collection<Appointment> getAppointments() {
024        return appointments;
025    }
026    @Override
027    public void addAppointment(Appointment t) {
028        appointments.add(t);
029
030    }
031
032
033public void setOwnerName(String ownerName) {
034    this.ownerName = ownerName;
035}
036}
