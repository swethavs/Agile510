036    /**
037     *This method takes the arguments passed form command line. It checks if the input string is options or arguments.
038     * The passed arguments for sppointments are added to appointmentBook.
039     * It also checks if the beginTime and endTime are in correct format.
040     * In any error scenario case, and error message is shown and program exits.
041     * @param args
042     * @param appointmentBook
043     * @throws ParseException
044     */
045    public static void prepareAppointmentBook(String[] args, AppointmentBook appointmentBook) throws ParseException {
046        Appointment appointment = new Appointment();
047
048        if(args.length==0){
049            System.out.println("You did not provide any arguments. Please provide the missing arguments in format: "+Constants.usage);
050        }else if(Arrays.asList(args).contains("-README")){
051            displayReadme();
052        } else {
053            for (int i = 0; i < args.length; i++) {
054                if (args[i].startsWith("-")) {
055                    if (args[i].contains(Constants.PRINT)) {
056                        System.out.println("Printing the Appointment : " + args[2]);
057                        i++;
058                    } else {
059                        System.out.println("Please provide the correct option.");
060                        System.exit(1);
061                    }
062                }
063                if ( args.length - i == 6) {
064                    appointment.setOwner(validateOwnerName(args[i])); //Checks if the argument starts with "-", it is not considered as owner name
065                    appointment.setDescription(checkNull(args[++i], "description")); //
066                    appointment.setBeginTimeString(checkDateTimeFormat(checkNull(args[++i], "beginDateTime").concat(" ").concat(checkNull(args[++i], "beginDateTime"))));
067                    appointment.setEndTimeString(checkDateTimeFormat(checkNull(args[++i], "endDateTime").concat(" ").concat(checkNull(args[++i], "endDateTime"))));
068                    compareEndDateBeginDate(appointment);
069                    appointmentBook.addAppointment(appointment);
070                    System.out.println(appointment.getOwner());
071                    System.out.println(appointment.getDescription());
072                    System.out.println(appointment.getBeginTimeString());
073                    System.out.println(appointment.getEndTimeString());
074                } else if( args.length - i < 6){
075                    System.out.println("Some arguments are missing. Please provide complete arguments : owner description beginDateTime endDateTime");
076                    System.exit(1);
077                }else if( args.length - i > 6){
078                    System.out.println("There are extra arguments provided than required arguments.\n" +
079                            "Please provide the arguments in order: owner description beginDateTime endDateTime");
080                    System.exit(1);
081                }
082            }
083
084        }
085    }

123    /**
124     * *This method checks for the correct format of date and time.
125     *The date and time should be in format "mm/dd/yyyy hh:mm' or "m/d/yyyy hh:mm".
126     * This method also check for invalid dates e.g. 13/01/2015 or 12/40/2015.
127     * Also enforces the year has to pass with 4 digits.
128     * @param value
129     * @return
130     * @throws ParseException
131     */
132    private static String checkDateTimeFormat(String value) throws ParseException {
133        if (value == null || !value.matches("^\\d{1,2}/\\d{1,2}/\\d{4} \\d{1,2}:\\d{2}$")) {
134            System.out.println("Please provide the date and time in format mm/dd/yyyy hh:mm");
135            System.exit(1);
136        }else{
137            SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy HH:mm");
138            df.setLenient(false);
139           df.parse(value);
140        }
141        return value;
142    }
